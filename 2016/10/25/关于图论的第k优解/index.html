关于图论的第2优解

正常思路是spfa找最短路，再拆最短路上的边走spfa，保存最小值。

(本来我准备以玛丽卡为例题，鬼知道它是拆完边后保存最大值，讲道理，就应该是最坏情况)

正如最近的博文所述，也可以用多路归并解决。

#include<bits/stdc++.h>
using namespace std;

const int maxN=1000+5,maxM=1000000+5,maxK=5,INF=0x7F7F7F7F;
struct Edge { int a,b,dist; };
vector G[maxN];
vector edges;
queue Q;
bool inq[maxN][maxN];
int f[maxN][maxK];
int K=2,T[maxK],N,M;
int q1,q2,a1,a2;
void merge(int*,int*,int);

int main()
{
    int u,v,d,now,ans=0;
    scanf("%d %d",&N,&M);
    for(int i=1;i<=N;i++)
        for(int j=1;j<=K;j++)
            f[i][j]=INF;
    f[1][1]=0;
    for(int i=0;i<M;i++)
    {
        scanf("%d %d %d",&u,&v,&d);
        edges.push_back((Edge){u,v,d});
        G[u].push_back(i);G[v].push_back(i);
    }
    Q.push(1);
    while(!Q.empty())
    {
        now=Q.front();Q.pop();
        for(int i=G[now].size()-1;i>=0;i--)
        {
            Edge& e=edges[G[now][i]];
            if(now==e.a) v=e.b;else v=e.a;
            if(inq[now][v]) continue;
            inq[now][v]=1;
            Q.push(v);
            merge(f[v],f[now],e.dist);
        }
    }
    cout<<f[N][K]<<endl;
    return 0;
}

void merge(int* A,int* B,int C)
{
    q1=1,q2=1;
    for(int i=1;i<=K;i++)
    {
        a1=A[q1],a2=B[q2]+C;
        if(a1>a2) T[i]=a2,q2++;
        else T[i]=a1,q1++;
    }
    for(int i=1;i<=K;i++) A[i]=T[i];
}

上述代码无法保证正确，但我相信算法才是最重要的。
